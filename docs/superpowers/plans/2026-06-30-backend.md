# Backend (Schema, Auth, Sharing, Sync) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the domain backend on top of the Fastify+SQLite skeleton: schema, anonymous cookie-auth with sliding renewal, device linking, list sharing with access isolation, and the offline-first `/lists/:id/sync` protocol with whole-item LWW + tombstones + idempotency, plus frequency-based suggestions.

**Architecture:** One SQLite file, opened in `db.ts`. Schema + category seeds live in `schema.ts` and are applied by `buildApp` (so in-memory test DBs are initialized too). Cross-cutting auth (`@fastify/cookie` + `onRequest` hook resolving `request.accountId` + sliding `Set-Cookie`) lives in `auth.ts`. Routes are grouped by responsibility into `routes/accounts.ts` (identity + categories), `routes/lists.ts` (lists CRUD + sharing), `routes/sync.ts` (sync + suggestions). Small reusable helpers: `ids.ts` (id/token/code/normalize), `map.ts` (row→domain), `access.ts` (membership). Source of truth is the server; clients generate `item`/`list` ids. Sync is whole-item LWW arbitrated by server `seq`: changes are field-patches applied column-wise, so edits to different fields both survive and same-field edits are won by the last change to reach the server.

**Tech Stack:** Fastify v5, better-sqlite3, `@fastify/cookie`, `node:crypto`, `node:test`, tsx. `@/` path alias → `packages/server/src`.

```typescript
import { randomUUID, randomBytes } from "node:crypto";

export const newId = (): string => randomUUID();

export const newToken = (): string => randomBytes(32).toString("base64url");

// Без 0/O/1/I — чтобы код читался вслух без путаницы.
const ALPHABET = "ABCDEFGHJKLMNPQRSTUVWXYZ23456789";

export function newCode(len = 6): string {
  const bytes = randomBytes(len);
  let s = "";
  for (let i = 0; i < len; i++) {
    // ponytail: лёгкий modulo-bias на 32-символьном алфавите пренебрежим
    s += ALPHABET[bytes[i]! % ALPHABET.length];
  }
  return s;
}

export const normalizeName = (s: string): string => s.trim().toLowerCase();
```

- [ ] **Step 2: Write `packages/server/src/auth.ts`**

`registerAuth` registers the cookie plugin and an `onRequest` hook. Public routes skip auth. For everything else the hook resolves the device token from the cookie to `request.accountId`, returns 401 if missing/invalid, and re-sets the cookie with a fresh `Max-Age` (sliding renewal — header only, no DB write). `setAuthCookie` is used by the accounts/link routes to issue the cookie on a new device.

```typescript
import type { FastifyInstance, FastifyReply } from "fastify";
import cookie from "@fastify/cookie";

export const COOKIE = "kupi_dt";
const MAX_AGE = 400 * 24 * 60 * 60; // 400 дней — cap Chrome для Max-Age

const cookieOpts = () =>
  ({ httpOnly: true, secure: true, sameSite: "lax", path: "/", maxAge: MAX_AGE }) as const;

export function setAuthCookie(reply: FastifyReply, token: string): void {
  reply.setCookie(COOKIE, token, cookieOpts());
}

const PUBLIC = new Set(["/health", "/accounts", "/link"]);

export function registerAuth(app: FastifyInstance): void {
  app.register(cookie);
  app.decorateRequest("accountId", "");

  app.addHook("onRequest", async (req, reply) => {
    const path = req.url.split("?")[0];
    if (PUBLIC.has(path)) return;

    const token = req.cookies[COOKIE];
    const device = token
      ? (app.db.prepare("SELECT account_id FROM devices WHERE token = ?").get(token) as
          | { account_id: string }
          | undefined)
      : undefined;

    if (!device) {
      return reply.code(401).send({ error: "unauthorized" });
    }

    req.accountId = device.account_id;
    setAuthCookie(reply, token!); // sliding: продлеваем TTL куки, значение то же
  });
}

declare module "fastify" {
  interface FastifyRequest {
    accountId: string;
  }
}
```

- [ ] **Step 1: Write `packages/server/src/bootstrap.ts`**

Shared by the accounts and link routes: assemble the payload a freshly-authenticated device needs (its account, all lists it belongs to, the category preset).

```typescript
import type { Bootstrap } from "@kupi/shared";
import type { Db } from "@/db";
import { rowToAccount, rowToCategory, rowToList } from "@/map";

export function buildBootstrap(db: Db, accountId: string): Bootstrap {
  const account = rowToAccount(db.prepare("SELECT * FROM accounts WHERE id = ?").get(accountId));
  const lists = (
    db
      .prepare(
        `SELECT l.* FROM lists l
         JOIN list_members m ON m.list_id = l.id
         WHERE m.account_id = ?
         ORDER BY l.created_at`,
      )
      .all(accountId) as any[]
  ).map(rowToList);
  const categories = (
    db.prepare("SELECT * FROM categories ORDER BY sort").all() as any[]
  ).map(rowToCategory);
  return { account, lists, categories };
}
```

- [ ] **Step 2: Write `packages/server/src/routes/accounts.ts`**

`POST /accounts` creates an account + first device + a default "Мои покупки" list (owner membership), sets the cookie, returns the bootstrap. `GET /categories` returns the preset.

```typescript
import type { FastifyInstance } from "fastify";
import { newId, newToken } from "@/ids";
import { setAuthCookie } from "@/auth";
import { buildBootstrap } from "@/bootstrap";
import { rowToCategory } from "@/map";

export function accountRoutes(app: FastifyInstance): void {
  app.post("/accounts", async (_req, reply) => {
    const now = Date.now();
    const accountId = newId();
    const token = newToken();
    const deviceId = newId();
    const listId = newId();

    app.db.transaction(() => {
      app.db.prepare("INSERT INTO accounts (id, created_at) VALUES (?, ?)").run(accountId, now);
      app.db
        .prepare("INSERT INTO devices (id, account_id, token, created_at) VALUES (?, ?, ?, ?)")
        .run(deviceId, accountId, token, now);
      app.db
        .prepare(
          "INSERT INTO lists (id, name, owner_account_id, seq, created_at) VALUES (?, ?, ?, 0, ?)",
        )
        .run(listId, "Мои покупки", accountId, now);
      app.db
        .prepare("INSERT INTO list_members (list_id, account_id, role) VALUES (?, ?, 'owner')")
        .run(listId, accountId);
    })();

    setAuthCookie(reply, token);
    return reply.code(201).send(buildBootstrap(app.db, accountId));
  });

  app.get("/categories", async () => {
    return (app.db.prepare("SELECT * FROM categories ORDER BY sort").all() as any[]).map(
      rowToCategory,
    );
  });
}
```

- [ ] **Step 1: Write `packages/server/src/routes/link.ts`**

`POST /link-codes` (authenticated — device A) mints a short single-use code (TTL 10 min). `POST /link` (public — device B) redeems it: creates a new device row for the same account, issues B its own cookie, returns the shared bootstrap. Codes are single-use (deleted on redemption).

```typescript
import type { FastifyInstance } from "fastify";
import { newCode, newId, newToken } from "@/ids";
import { setAuthCookie } from "@/auth";
import { buildBootstrap } from "@/bootstrap";

const LINK_TTL = 10 * 60 * 1000;

export function linkRoutes(app: FastifyInstance): void {
  app.post("/link-codes", async (req) => {
    const code = newCode(6);
    const expiresAt = Date.now() + LINK_TTL;
    app.db
      .prepare("INSERT INTO link_codes (code, account_id, expires_at) VALUES (?, ?, ?)")
      .run(code, req.accountId, expiresAt);
    return { code };
  });

  app.post<{ Body: { code?: string } }>(
    "/link",
    {
      schema: {
        body: {
          type: "object",
          required: ["code"],
          properties: { code: { type: "string" } },
        },
      },
    },
    async (req, reply) => {
      const code = req.body.code!;
      const row = app.db
        .prepare("SELECT account_id, expires_at FROM link_codes WHERE code = ?")
        .get(code) as { account_id: string; expires_at: number } | undefined;

      if (!row || row.expires_at < Date.now()) {
        return reply.code(400).send({ error: "invalid_code" });
      }

      const token = newToken();
      const deviceId = newId();
      app.db.transaction(() => {
        app.db
          .prepare("INSERT INTO devices (id, account_id, token, created_at) VALUES (?, ?, ?, ?)")
          .run(deviceId, row.account_id, token, Date.now());
        app.db.prepare("DELETE FROM link_codes WHERE code = ?").run(code); // одноразовый
      })();

      setAuthCookie(reply, token);
      return buildBootstrap(app.db, row.account_id);
    },
  );
}
```

- [ ] **Step 2: Write `packages/server/src/routes/lists.ts`**

`GET /lists`, `POST /lists` (creates list + owner membership), `PATCH /lists/:id` (rename, members only), `POST /lists/:id/invites` (owner only, reusable code with TTL), `POST /lists/join` (authenticated joiner adds a member membership).

```typescript
import type { FastifyInstance } from "fastify";
import { newCode, newId } from "@/ids";
import { isMember, isOwner } from "@/access";
import { rowToList } from "@/map";

const INVITE_TTL = 7 * 24 * 60 * 60 * 1000;

export function listRoutes(app: FastifyInstance): void {
  app.get("/lists", async (req) => {
    return (
      app.db
        .prepare(
          `SELECT l.* FROM lists l
           JOIN list_members m ON m.list_id = l.id
           WHERE m.account_id = ?
           ORDER BY l.created_at`,
        )
        .all(req.accountId) as any[]
    ).map(rowToList);
  });

  app.post<{ Body: { name?: string } }>(
    "/lists",
    {
      schema: {
        body: {
          type: "object",
          required: ["name"],
          properties: { name: { type: "string", minLength: 1 } },
        },
      },
    },
    async (req, reply) => {
      const id = newId();
      const now = Date.now();
      app.db.transaction(() => {
        app.db
          .prepare(
            "INSERT INTO lists (id, name, owner_account_id, seq, created_at) VALUES (?, ?, ?, 0, ?)",
          )
          .run(id, req.body.name, req.accountId, now);
        app.db
          .prepare("INSERT INTO list_members (list_id, account_id, role) VALUES (?, ?, 'owner')")
          .run(id, req.accountId);
      })();
      return reply.code(201).send(rowToList(app.db.prepare("SELECT * FROM lists WHERE id = ?").get(id)));
    },
  );

  app.patch<{ Params: { id: string }; Body: { name?: string } }>(
    "/lists/:id",
    {
      schema: {
        body: {
          type: "object",
          required: ["name"],
          properties: { name: { type: "string", minLength: 1 } },
        },
      },
    },
    async (req, reply) => {
      if (!isMember(app.db, req.params.id, req.accountId)) {
        return reply.code(404).send({ error: "not_found" });
      }
      app.db.prepare("UPDATE lists SET name = ? WHERE id = ?").run(req.body.name, req.params.id);
      return rowToList(app.db.prepare("SELECT * FROM lists WHERE id = ?").get(req.params.id));
    },
  );

  app.post<{ Params: { id: string } }>("/lists/:id/invites", async (req, reply) => {
    if (!isOwner(app.db, req.params.id, req.accountId)) {
      return reply.code(404).send({ error: "not_found" });
    }
    const code = newCode(8);
    app.db
      .prepare("INSERT INTO list_invites (code, list_id, expires_at) VALUES (?, ?, ?)")
      .run(code, req.params.id, Date.now() + INVITE_TTL);
    return { code };
  });

  app.post<{ Body: { code?: string } }>(
    "/lists/join",
    {
      schema: {
        body: {
          type: "object",
          required: ["code"],
          properties: { code: { type: "string" } },
        },
      },
    },
    async (req, reply) => {
      const inv = app.db
        .prepare("SELECT list_id, expires_at FROM list_invites WHERE code = ?")
        .get(req.body.code) as { list_id: string; expires_at: number } | undefined;
      if (!inv || inv.expires_at < Date.now()) {
        return reply.code(400).send({ error: "invalid_code" });
      }
      app.db
        .prepare("INSERT OR IGNORE INTO list_members (list_id, account_id, role) VALUES (?, ?, 'member')")
        .run(inv.list_id, req.accountId);
      return rowToList(app.db.prepare("SELECT * FROM lists WHERE id = ?").get(inv.list_id));
    },
  );
}
```

- [ ] **Step 1: Write `packages/server/src/merge.ts`**

The per-change merge. Idempotency via `applied_ops`. Each applied change bumps `list.seq` and stamps `item.version = seq`. Remove-wins: a tombstoned item is never resurrected by a non-delete change. Upserts patch only the sent fields (column-wise via `COALESCE`), so concurrent edits to different fields both survive and same-field edits are won by the last change to reach the server. Frequency is incremented when an upsert creates a new named item.

```typescript
import type { ItemChange } from "@kupi/shared";
import type { Db } from "@/db";
import { normalizeName } from "@/ids";

function bumpSeq(db: Db, listId: string): number {
  db.prepare("UPDATE lists SET seq = seq + 1 WHERE id = ?").run(listId);
  return (db.prepare("SELECT seq FROM lists WHERE id = ?").get(listId) as { seq: number }).seq;
}

function incFreq(db: Db, accountId: string, name: string): void {
  const n = normalizeName(name);
  if (!n) return;
  db.prepare(
    `INSERT INTO item_frequency (account_id, normalized_name, count) VALUES (?, ?, 1)
     ON CONFLICT(account_id, normalized_name) DO UPDATE SET count = count + 1`,
  ).run(accountId, n);
}

export function applyChange(db: Db, listId: string, accountId: string, ch: ItemChange): void {
  // идемпотентность: повторная доставка одного clientOpId не применяется дважды
  const ins = db
    .prepare("INSERT OR IGNORE INTO applied_ops (client_op_id) VALUES (?)")
    .run(ch.clientOpId);
  if (ins.changes === 0) return;

  const cur = db.prepare("SELECT * FROM items WHERE id = ?").get(ch.itemId) as
    | { deleted: number }
    | undefined;

  // remove-wins: удалённое не воскрешаем правкой
  if (cur && cur.deleted && ch.op !== "delete") return;

  const seq = bumpSeq(db, listId);
  const now = Date.now();
  const f = ch.fields ?? {};

  if (!cur) {
    db.prepare(
      `INSERT INTO items (id, list_id, name, quantity, category_id, checked, version, deleted, updated_at)
       VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)`,
    ).run(
      ch.itemId,
      listId,
      f.name ?? "",
      f.quantity ?? 1,
      f.categoryId ?? null,
      f.checked ? 1 : 0,
      seq,
      ch.op === "delete" ? 1 : 0,
      now,
    );
    if (ch.op !== "delete" && f.name) incFreq(db, accountId, f.name);
    return;
  }

  if (ch.op === "delete") {
    db.prepare("UPDATE items SET deleted = 1, version = ?, updated_at = ? WHERE id = ?").run(
      seq,
      now,
      ch.itemId,
    );
    return;
  }

  // патч: COALESCE(@x, col) — null (поле не прислано) оставляет текущее значение
  db.prepare(
    `UPDATE items SET
       name = COALESCE(@name, name),
       quantity = COALESCE(@quantity, quantity),
       category_id = COALESCE(@category_id, category_id),
       checked = COALESCE(@checked, checked),
       version = @seq, updated_at = @now
     WHERE id = @id`,
  ).run({
    name: f.name ?? null,
    quantity: f.quantity ?? null,
    category_id: f.categoryId ?? null,
    checked: f.checked === undefined ? null : f.checked ? 1 : 0,
    seq,
    now,
    id: ch.itemId,
  });
}
```

> **ponytail note:** clearing a category (`categoryId: null`) is treated as "unchanged" here, because `COALESCE` can't tell "absent" from "explicit null" — clearing a category isn't an MVP feature. Add a sentinel if it ever is.

- [ ] **Step 2: Write `packages/server/src/routes/sync.ts`**

`POST /lists/:id/sync` (member-only) applies changes in one transaction, then returns the new `seq` and every item with `version > lastSeenSeq` (tombstones included, so deletes propagate). `GET /suggestions?q=` returns the caller's most-used names by prefix.

```typescript
import type { FastifyInstance } from "fastify";
import type { SyncRequest, SyncResponse } from "@kupi/shared";
import { isMember } from "@/access";
import { applyChange } from "@/merge";
import { rowToItem } from "@/map";
import { normalizeName } from "@/ids";

const syncSchema = {
  body: {
    type: "object",
    required: ["lastSeenSeq", "changes"],
    properties: {
      lastSeenSeq: { type: "integer", minimum: 0 },
      changes: {
        type: "array",
        items: {
          type: "object",
          required: ["itemId", "clientOpId", "op"],
          properties: {
            itemId: { type: "string" },
            clientOpId: { type: "string" },
            op: { enum: ["upsert", "delete"] },
            fields: { type: "object" },
          },
        },
      },
    },
  },
};

export function syncRoutes(app: FastifyInstance): void {
  app.post<{ Params: { id: string }; Body: SyncRequest }>(
    "/lists/:id/sync",
    { schema: syncSchema },
    async (req, reply): Promise<SyncResponse> => {
      const listId = req.params.id;
      if (!isMember(app.db, listId, req.accountId)) {
        return reply.code(404).send({ error: "not_found" }) as never;
      }

      const { lastSeenSeq, changes } = req.body;
      app.db.transaction(() => {
        for (const ch of changes) applyChange(app.db, listId, req.accountId, ch);
      })();

      const seq = (app.db.prepare("SELECT seq FROM lists WHERE id = ?").get(listId) as {
        seq: number;
      }).seq;
      const items = (
        app.db
          .prepare("SELECT * FROM items WHERE list_id = ? AND version > ? ORDER BY version")
          .all(listId, lastSeenSeq) as any[]
      ).map(rowToItem);

      return { seq, items };
    },
  );

  app.get<{ Querystring: { q?: string } }>("/suggestions", async (req) => {
    const q = normalizeName(req.query.q ?? "");
    if (!q) return [];
    return app.db
      .prepare(
        `SELECT normalized_name AS name, count FROM item_frequency
         WHERE account_id = ? AND normalized_name LIKE ?
         ORDER BY count DESC, normalized_name
         LIMIT 10`,
      )
      .all(req.accountId, `${q}%`);
  });
}
```
